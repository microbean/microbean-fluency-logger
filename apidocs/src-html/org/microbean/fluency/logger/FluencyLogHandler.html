017package org.microbean.fluency.logger;
018
019import java.io.IOException;
020
021import java.util.Arrays;
022import java.util.Collection;
023import java.util.HashMap;
024import java.util.Map;
025import java.util.Objects;
026
027import java.util.logging.ErrorManager;
028import java.util.logging.Formatter;
029import java.util.logging.Handler;
030import java.util.logging.LogRecord;
031import java.util.logging.SimpleFormatter;
032
033import org.komamitsu.fluency.Fluency;
034
035import org.komamitsu.fluency.fluentd.FluencyBuilderForFluentd;
036
037public class FluencyLogHandler extends Handler {
038
039  private final Fluency fluency;
040
041  public FluencyLogHandler() {
042    this(new FluencyBuilderForFluentd().build());
043  }
044
045  public FluencyLogHandler(final Fluency fluency) {
046    super();
047    this.fluency = Objects.requireNonNull(fluency);
048    this.setFormatter(new SimpleFormatter());
049  }
050
051  @Override
052  public void close() {
053    try {
054      this.fluency.close();
055    } catch (final IOException | RuntimeException exception) {
056      this.reportError(exception.getMessage(), exception, ErrorManager.CLOSE_FAILURE);
057    }
058  }
059
060  @Override
061  public void flush() {
062    try {
063      this.fluency.flush();
064    } catch (final IOException | RuntimeException exception) {
065      this.reportError(exception.getMessage(), exception, ErrorManager.FLUSH_FAILURE);
066    }
067  }
068
069  @Override
070  public void publish(final LogRecord logRecord) {
071    if (this.isLoggable(logRecord)) {
072      final Map<String, Object> event = this.toMap(logRecord);
073      if (event != null && !event.isEmpty()) {
074        final long logRecordSecondsSinceEpoch = logRecord.getMillis() / 1000L;
075        try {
076          this.fluency.emit(logRecord.getLoggerName(), logRecordSecondsSinceEpoch, event);
077        } catch (final IOException | RuntimeException exception) {
078          this.reportError(exception.getMessage(), exception, ErrorManager.WRITE_FAILURE);
079        }
080      }
081    }
082  }
083
084  protected Map<String, Object> toMap(final LogRecord logRecord) {
085    final Map<String, Object> event = new HashMap<>();
086    if (logRecord != null) {
087      final Formatter formatter = this.getFormatter();
088      final String message;
089      if (formatter == null) {
090        message = logRecord.getMessage();
091      } else {
092        message = formatter.format(logRecord);
093      }
094      final Object[] parametersArray = logRecord.getParameters();
095      final Collection<?> parameters;
096      if (parametersArray == null || parametersArray.length <= 0) {
097        parameters = null;
098      } else {
099        parameters = Arrays.asList(parametersArray);
100      }
101      event.put("level", logRecord.getLevel());
102      event.put("message", message);
103      event.put("parameters", parameters);
104      event.put("sequenceNumber", Long.valueOf(logRecord.getSequenceNumber()));
105      event.put("sourceClassName", logRecord.getSourceClassName());
106      event.put("sourceMethodName", logRecord.getSourceMethodName());
107      event.put("threadID", logRecord.getThreadID());
108      event.put("thrown", logRecord.getThrown());
109    }
110    return event;
111  }
112
113}
